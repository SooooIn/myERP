create table t_login(id varchar(30), pass varchar(30), primary key(id));

create table position(val varchar(15), code varchar(10), primary key(code));

create table department(val varchar(15), code varchar(10), primary key(code));

create table religion(val varchar(15), code varchar(10), primary key(code));

create table retirement_reason(val varchar(15));

create table member(name_k varchar(10), name_e varchar(20), name_c varchar(10),
					m_address varchar(50), marriage int, e_mail varchar(30),
					c_number varchar(30), sex int, height int, weight_ int,
					blood varchar(10), sight_l varchar(10), sight_r varchar(10),
					religioin varchar(15), depart varchar(10), position varchar(10),
					birth_date varchar(30), join_date varchar(30), id varchar(30), work int
					primary key(id));

create table family(f_id varchar(30), relation varchar(10), place varchar(15),
					f_name varchar(10), job varchar(10), birth_date varchar(30), c_number varchar(30));

create table retired(id varchar(30), r_name varchar(10), c_number varchar(30),
					depart varchar(10), position varchar(10), retire_date varchar(30), r_reason varchar(15)
					primary key(id));
